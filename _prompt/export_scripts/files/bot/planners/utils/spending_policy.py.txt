# bot/planners/utils/spending_policy.py

from __future__ import annotations

from dataclasses import dataclass

from bot.mind.attention import Attention
from bot.mind.awareness import Awareness, K


@dataclass(frozen=True)
class SpendingPolicyDecision:
    mode: str
    target_refineries: int
    target_refineries_default: int
    target_workers_per_refinery: int
    gas_stock: int
    mineral_stock: int
    gas_net_per_s: float
    mineral_net_per_s: float
    gas_income_per_s: float
    gas_spend_per_s: float
    mineral_income_per_s: float
    mineral_spend_per_s: float
    gas_mix: float
    imbalance_slope: float
    changed_at: float


class SpendingPolicy:
    """
    Economic arbitration for worker gas/mineral split.
    Uses stock + flow so macro execution reacts to sustained imbalance.
    """

    def __init__(self, *, flow_alpha: float = 0.35):
        self.flow_alpha = float(max(0.05, min(0.95, flow_alpha)))

    def step(
        self,
        *,
        attention: Attention,
        awareness: Awareness,
        now: float,
        lag_tech: float,
        lag_spend: float,
        lag_prod: float,
        cfg,
    ) -> SpendingPolicyDecision:
        bases = int(attention.macro.bases_total)
        gas_stock = int(attention.economy.gas)
        mineral_stock = int(attention.economy.minerals)
        target_refineries_default = max(0, int(bases) * 2)
        target_refineries = int(target_refineries_default)
        workers_per_refinery = int(getattr(cfg, "gas_target_workers_default", 3))
        mode = "normal"

        prev = awareness.mem.get(K("control", "resource", "flow", "prev"), now=now, default={}) or {}
        prev_t = float(prev.get("t", now) or now)
        prev_m = float(prev.get("minerals", mineral_stock) or mineral_stock)
        prev_g = float(prev.get("gas", gas_stock) or gas_stock)
        prev_m_income_ema = float(prev.get("m_income_ema", 0.0) or 0.0)
        prev_m_spend_ema = float(prev.get("m_spend_ema", 0.0) or 0.0)
        prev_g_income_ema = float(prev.get("g_income_ema", 0.0) or 0.0)
        prev_g_spend_ema = float(prev.get("g_spend_ema", 0.0) or 0.0)
        prev_imbalance = float(prev.get("imbalance", 0.0) or 0.0)

        dt = max(1e-3, min(2.0, float(now) - float(prev_t)))
        m_net = (float(mineral_stock) - float(prev_m)) / dt
        g_net = (float(gas_stock) - float(prev_g)) / dt
        m_income = max(0.0, m_net)
        m_spend = max(0.0, -m_net)
        g_income = max(0.0, g_net)
        g_spend = max(0.0, -g_net)

        alpha = float(self.flow_alpha)
        m_income_ema = (1.0 - alpha) * prev_m_income_ema + (alpha * m_income)
        m_spend_ema = (1.0 - alpha) * prev_m_spend_ema + (alpha * m_spend)
        g_income_ema = (1.0 - alpha) * prev_g_income_ema + (alpha * g_income)
        g_spend_ema = (1.0 - alpha) * prev_g_spend_ema + (alpha * g_spend)
        spend_sum = max(1e-3, m_spend_ema + g_spend_ema)
        gas_mix = g_spend_ema / spend_sum

        target_m = max(1.0, float(getattr(cfg, "bank_target_minerals", 650)))
        target_g = max(1.0, float(getattr(cfg, "bank_target_gas", 220)))
        norm_m = float(mineral_stock) / target_m
        norm_g = float(gas_stock) / target_g
        imbalance = norm_g - norm_m
        imbalance_slope = (float(imbalance) - float(prev_imbalance)) / dt

        tech_pressure = max(0.0, float(lag_tech))
        expand_pressure = max(0.0, float(lag_spend))
        production_pressure = max(0.0, float(lag_prod))

        if gas_stock >= int(cfg.gas_overflow_hard) and mineral_stock <= int(cfg.mineral_low_soft) and tech_pressure < 0.55:
            workers_per_refinery = 0
            target_refineries = max(0, target_refineries - 2)
            mode = "gas_overflow_hard"
        elif gas_stock >= int(cfg.gas_overflow_soft) and mineral_stock <= int(cfg.mineral_low_soft) and tech_pressure < 0.65:
            workers_per_refinery = 1
            target_refineries = max(0, target_refineries - 1)
            mode = "gas_overflow_soft"
        elif (
            gas_stock >= int(cfg.gas_ratio_min_stock_hard)
            and gas_stock >= int(float(cfg.gas_to_mineral_ratio_hard) * max(1.0, float(mineral_stock)))
            and tech_pressure < 0.75
        ):
            workers_per_refinery = 0
            target_refineries = max(0, target_refineries - 2)
            mode = "gas_ratio_hard"
        elif (
            gas_stock >= int(cfg.gas_ratio_min_stock_soft)
            and gas_stock >= int(float(cfg.gas_to_mineral_ratio_soft) * max(1.0, float(mineral_stock)))
            and tech_pressure < 0.85
        ):
            workers_per_refinery = 1
            target_refineries = max(0, target_refineries - 1)
            mode = "gas_ratio_soft"

        flow_shift_soft = (
            production_pressure >= 0.65
            and tech_pressure < 0.70
            and gas_stock >= int(cfg.gas_overflow_soft)
            and g_net > 0.0
            and imbalance_slope >= 0.0
            and gas_mix < 0.45
            and production_pressure > expand_pressure
        )
        flow_shift_hard = (
            production_pressure >= 0.90
            and tech_pressure < 0.65
            and gas_stock >= int(cfg.gas_overflow_hard)
            and g_net > 0.0
            and imbalance_slope > 0.03
            and gas_mix < 0.40
        )
        if flow_shift_hard:
            workers_per_refinery = min(workers_per_refinery, 0)
            target_refineries = max(0, target_refineries - 2)
            mode = "flow_shift_hard_to_minerals"
        elif flow_shift_soft and mode == "normal":
            workers_per_refinery = min(workers_per_refinery, 1)
            target_refineries = max(0, target_refineries - 1)
            mode = "flow_shift_soft_to_minerals"

        if tech_pressure >= 0.70:
            workers_per_refinery = max(2, int(workers_per_refinery))
            mode = "tech_pressure"

        workers_per_refinery = max(int(cfg.gas_target_workers_min), min(3, int(workers_per_refinery)))

        prev_status = awareness.mem.get(K("macro", "gas", "status"), now=now, default={}) or {}
        prev_mode = str(prev_status.get("mode", "") or "")
        prev_workers = int(prev_status.get("target_workers_per_refinery", workers_per_refinery) or workers_per_refinery)
        prev_refineries = int(prev_status.get("target_refineries", target_refineries) or target_refineries)
        prev_changed_at = float(prev_status.get("changed_at", now) or now)
        dt_change = max(0.0, float(now) - float(prev_changed_at))
        changed = (int(prev_workers) != int(workers_per_refinery)) or (int(prev_refineries) != int(target_refineries))
        emergency_modes = {"gas_overflow_hard", "gas_ratio_hard", "tech_pressure", "flow_shift_hard_to_minerals"}
        if changed and mode not in emergency_modes:
            if dt_change < float(cfg.gas_mode_hold_s):
                workers_per_refinery = int(prev_workers)
                target_refineries = int(prev_refineries)
                mode = str(prev_mode or mode)
                changed = False
            else:
                if int(prev_workers) != int(workers_per_refinery) and dt_change < float(cfg.gas_workers_change_cooldown_s):
                    workers_per_refinery = int(prev_workers)
                    mode = str(prev_mode or mode)
                    changed = False
                if int(prev_refineries) != int(target_refineries) and dt_change < float(cfg.gas_refineries_change_cooldown_s):
                    target_refineries = int(prev_refineries)
                    mode = str(prev_mode or mode)
                    changed = False
        changed_at = float(now) if changed else float(prev_changed_at)

        awareness.mem.set(
            K("control", "resource", "flow", "prev"),
            value={
                "t": float(now),
                "minerals": int(mineral_stock),
                "gas": int(gas_stock),
                "m_income_ema": float(m_income_ema),
                "m_spend_ema": float(m_spend_ema),
                "g_income_ema": float(g_income_ema),
                "g_spend_ema": float(g_spend_ema),
                "imbalance": float(imbalance),
            },
            now=now,
            ttl=20.0,
        )

        return SpendingPolicyDecision(
            mode=str(mode),
            target_refineries=int(target_refineries),
            target_refineries_default=int(target_refineries_default),
            target_workers_per_refinery=int(workers_per_refinery),
            gas_stock=int(gas_stock),
            mineral_stock=int(mineral_stock),
            gas_net_per_s=float(g_net),
            mineral_net_per_s=float(m_net),
            gas_income_per_s=float(g_income_ema),
            gas_spend_per_s=float(g_spend_ema),
            mineral_income_per_s=float(m_income_ema),
            mineral_spend_per_s=float(m_spend_ema),
            gas_mix=float(gas_mix),
            imbalance_slope=float(imbalance_slope),
            changed_at=float(changed_at),
        )
