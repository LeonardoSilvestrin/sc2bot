from __future__ import annotations

from dataclasses import dataclass

from sc2.ids.unit_typeid import UnitTypeId as U


@dataclass(frozen=True)
class OpeningTimingRule:
    # Human-readable id for logs/tuning.
    name: str
    # "ANY", "ZERG", "TERRAN", "PROTOSS"
    race: str
    structure: U
    # Active only inside this observation window.
    earliest_t: float = 0.0
    latest_t: float = 9999.0
    # Constraints over observed structure counts.
    min_total_count: int = 1
    max_total_count: int | None = None
    min_main_count: int = 0
    max_main_count: int | None = None
    # Minimum observed build progress for the structure type.
    min_progress: float = 0.0
    # Score contribution and whether it is a hard-rush marker.
    score: float = 0.0
    confirm: bool = False
