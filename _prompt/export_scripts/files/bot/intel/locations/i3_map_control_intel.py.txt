# bot/intel/locations/i3_map_control_intel.py

"""
Map Control Intel — fatos observáveis da natural e do terreno adjacente.

Responsabilidade: observar estado estrutural da nat (tomada, voando, fortified,
presença inimiga, geometria de posições defensáveis). NÃO decide postura.

Nota: should_secure e safe_to_land permanecem como fatos derivados simplificados
para compatibilidade com LandBaseTask e outros consumidores legados. A lógica de
DECISÃO de postura está em bot/intel/strategy/i3_army_posture_intel.py.
"""
from __future__ import annotations

from dataclasses import dataclass

from ares.consts import BuildingSize
from sc2.ids.unit_typeid import UnitTypeId as U
from sc2.position import Point2

from bot.mind.attention import Attention
from bot.mind.awareness import Awareness, K

_WORKER_TYPES = {U.SCV, U.PROBE, U.DRONE, U.MULE, U.LARVA, U.EGG}
_OWN_ARMY_WEIGHTS = {
    U.SIEGETANK: 3.0,
    U.SIEGETANKSIEGED: 4.5,
    U.WIDOWMINE: 2.0,
    U.WIDOWMINEBURROWED: 3.2,
    U.CYCLONE: 1.5,
    U.MARAUDER: 1.35,
    U.MARINE: 1.0,
    U.HELLION: 0.85,
    U.THOR: 3.2,
    U.THORAP: 3.2,
    U.MEDIVAC: 0.6,
    U.BUNKER: 4.0,
}
_ENEMY_WEIGHTS = {
    U.SIEGETANK: 3.0,
    U.SIEGETANKSIEGED: 4.5,
    U.WIDOWMINE: 2.0,
    U.WIDOWMINEBURROWED: 3.2,
    U.CYCLONE: 1.6,
    U.MARAUDER: 1.35,
    U.MARINE: 1.0,
    U.HELLION: 0.9,
    U.THOR: 3.2,
    U.THORAP: 3.2,
    U.BUNKER: 4.0,
    U.PHOTONCANNON: 3.6,
    U.SPINECRAWLER: 3.6,
}


@dataclass(frozen=True)
class MapControlIntelConfig:
    ttl_s: float = 6.0
    release_clear_s: float = 7.0
    recent_window_s: float = 55.0
    min_total_power: float = 6.0
    min_main_power: float = 4.0
    max_enemy_power_at_nat: float = 2.6
    max_enemy_presence_power_at_nat: float = 2.25
    enemy_macro_lead_visible_bases: int = 3
    enemy_macro_lead_gap: int = 2
    enemy_macro_lead_clear_s: float = 4.0
    enemy_macro_lead_min_total_power: float = 4.0
    enemy_macro_lead_min_main_power: float = 3.0
    enemy_macro_lead_max_enemy_power_at_nat: float = 3.1
    enemy_macro_lead_max_enemy_presence_power_at_nat: float = 2.75
    rush_refresh_scan_stale_s: float = 12.0
    rush_refresh_min_interval_s: float = 12.0
    delayed_natural_min_total_power: float = 3.5
    delayed_natural_min_main_power: float = 2.5
    delayed_natural_max_enemy_power_at_nat: float = 3.2
    delayed_natural_max_enemy_presence_power_at_nat: float = 2.8
    natural_release_clear_s: float = 3.5
    natural_release_min_own_nat_power: float = 4.5


def _point_payload(pos: Point2 | None) -> dict[str, float] | None:
    if pos is None:
        return None
    return {"x": float(pos.x), "y": float(pos.y)}


def _point_from_payload(payload) -> Point2 | None:
    if not isinstance(payload, dict):
        return None
    try:
        return Point2((float(payload.get("x", 0.0) or 0.0), float(payload.get("y", 0.0) or 0.0)))
    except Exception:
        return None


def _pathable(bot, pos: Point2) -> bool:
    try:
        return bool(bot.in_pathing_grid(pos))
    except Exception:
        return True


def _placements_dict(bot) -> dict:
    try:
        return dict(bot.mediator.get_placements_dict or {})
    except Exception:
        return {}


def _height(bot, pos: Point2 | None) -> float:
    if pos is None:
        return -9999.0
    try:
        return float(bot.get_terrain_z_height(pos))
    except Exception:
        try:
            return float(bot.get_terrain_height(pos))
        except Exception:
            return -9999.0


def _ramp_edge_depth(bot, *, pos: Point2, bottom: Point2 | None, lowground_h: float) -> float:
    if bottom is None:
        return 9999.0
    for dist in (0.5, 1.0, 1.5, 2.0, 2.5, 3.0, 3.5, 4.0):
        try:
            probe = pos.towards(bottom, dist)
        except Exception:
            continue
        probe_h = _height(bot, probe)
        if probe_h <= (float(lowground_h) + 0.25):
            return float(dist)
    return 9999.0


def _main_ramp_bunker_target(bot) -> Point2 | None:
    try:
        ramp = getattr(bot, "main_base_ramp", None)
        top = getattr(ramp, "top_center", None) if ramp is not None else None
        barracks_pos = getattr(ramp, "barracks_correct_placement", None) if ramp is not None else None
        bottom = getattr(ramp, "bottom_center", None) if ramp is not None else None
        start = getattr(bot, "start_location", None)
    except Exception:
        return None
    if top is None or bottom is None or start is None:
        return None
    placements = _placements_dict(bot)
    per_size = ((placements.get(start, {}) or {}).get(BuildingSize.THREE_BY_THREE, {}) or {})
    lowground_h = _height(bot, bottom)
    highground_h = max(_height(bot, top), _height(bot, barracks_pos), _height(bot, start))
    best: tuple[float, Point2] | None = None
    for pos, info in per_size.items():
        if not isinstance(info, dict):
            continue
        if not bool(info.get("is_wall", False)) or not bool(info.get("bunker", False)):
            continue
        try:
            if not bool(bot.mediator.can_place_structure(position=pos, structure_type=U.BUNKER)):
                continue
        except Exception:
            continue
        pos_h = _height(bot, pos)
        if pos_h <= (float(lowground_h) + 0.25):
            continue
        if pos_h < (float(highground_h) - 0.75):
            continue
        edge_depth = _ramp_edge_depth(bot, pos=pos, bottom=bottom, lowground_h=lowground_h)
        if edge_depth > 4.25:
            continue
        if float(pos.distance_to(bottom)) > 6.5 or float(pos.distance_to(top)) > 6.5:
            continue
        if float(pos.distance_to(start)) < (float(top.distance_to(start)) - 0.75):
            continue
        score = 0.0
        score += max(0.0, 11.0 - float(pos.distance_to(top))) * 3.0
        score += max(0.0, 11.0 - float(pos.distance_to(bottom))) * 2.5
        if barracks_pos is not None:
            score += max(0.0, 8.0 - float(pos.distance_to(barracks_pos))) * 1.0
        score += max(0.0, 3.5 - abs(float(edge_depth) - 2.25)) * 4.0
        if best is None or score > best[0]:
            best = (float(score), pos)
    return best[1] if best is not None else None


def _our_natural(bot) -> Point2 | None:
    try:
        return bot.mediator.get_own_nat
    except Exception:
        pass
    exps = list(getattr(bot, "expansion_locations_list", []) or [])
    if not exps:
        return None
    try:
        ordered = sorted(
            [p for p in exps if float(p.distance_to(bot.start_location)) > 2.0],
            key=lambda p: float(bot.start_location.distance_to(p)),
        )
    except Exception:
        ordered = exps
    return ordered[0] if ordered else None


def _main_townhall(bot):
    ths = list(getattr(bot, "townhalls", []) or [])
    if not ths:
        return None
    try:
        ths.sort(key=lambda th: float(th.distance_to(bot.start_location)))
    except Exception:
        return ths[0]
    return ths[0]


def _enemy_combat_near(bot, *, center: Point2, radius: float) -> list:
    out = []
    for unit in list(getattr(bot, "enemy_units", []) or []):
        try:
            if unit.type_id in _WORKER_TYPES:
                continue
            if float(unit.distance_to(center)) <= float(radius):
                out.append(unit)
        except Exception:
            continue
    for struct in list(getattr(bot, "enemy_structures", []) or []):
        try:
            if float(struct.distance_to(center)) > float(radius):
                continue
            if struct.type_id in _WORKER_TYPES:
                continue
            if struct.type_id in {U.COMMANDCENTER, U.ORBITALCOMMAND, U.PLANETARYFORTRESS, U.HATCHERY, U.LAIR, U.HIVE, U.NEXUS}:
                continue
            out.append(struct)
        except Exception:
            continue
    return out


def _own_power_near(bot, *, center: Point2, radius: float) -> float:
    total = 0.0
    for unit in list(getattr(bot, "units", []) or []) + list(getattr(bot, "structures", []) or []):
        try:
            weight = float(_OWN_ARMY_WEIGHTS.get(unit.type_id, 0.0))
            if weight <= 0.0:
                continue
            if not bool(getattr(unit, "is_ready", True)):
                continue
            if float(unit.distance_to(center)) <= float(radius):
                total += float(weight)
        except Exception:
            continue
    return float(total)


def _own_total_power(bot) -> float:
    total = 0.0
    for unit in list(getattr(bot, "units", []) or []) + list(getattr(bot, "structures", []) or []):
        try:
            if not bool(getattr(unit, "is_ready", True)):
                continue
            total += float(_OWN_ARMY_WEIGHTS.get(unit.type_id, 0.0))
        except Exception:
            continue
    return float(total)


def _enemy_power(units: list) -> float:
    total = 0.0
    for unit in list(units or []):
        try:
            total += float(_ENEMY_WEIGHTS.get(unit.type_id, 1.0))
        except Exception:
            total += 1.0
    return float(total)


def derive_map_control_intel(
    bot,
    *,
    awareness: Awareness,
    attention: Attention,
    now: float,
    cfg: MapControlIntelConfig = MapControlIntelConfig(),
) -> None:
    nat = _our_natural(bot)
    if nat is None:
        return

    try:
        enemy_main = bot.enemy_start_locations[0]
    except Exception:
        enemy_main = nat

    hold_pos = nat.towards(enemy_main, 4.5)
    staging_pos = nat.towards(bot.start_location, 3.0)
    if not _pathable(bot, hold_pos):
        hold_pos = nat
    if not _pathable(bot, staging_pos):
        staging_pos = nat

    main_th = _main_townhall(bot)
    main_th_tag = int(getattr(main_th, "tag", -1) or -1)
    main_threat_active = False
    for th in list(getattr(attention.combat, "base_threats", ()) or ()):
        try:
            if int(th.th_tag) == int(main_th_tag) and int(th.urgency) > 0:
                main_threat_active = True
                break
        except Exception:
            continue

    main_enemy = _enemy_combat_near(bot, center=bot.start_location, radius=18.0)
    nat_enemy = _enemy_combat_near(bot, center=nat, radius=18.0)
    nat_enemy_power = _enemy_power(nat_enemy)

    our_bases = awareness.mem.get(K("intel", "our_bases", "registry"), now=now, default={}) or {}
    if not isinstance(our_bases, dict):
        our_bases = {}
    nat_entry = dict(our_bases.get("NATURAL", {})) if isinstance(our_bases.get("NATURAL", {}), dict) else {}
    nat_state = str(nat_entry.get("state", "PLANNED") or "PLANNED").upper()
    nat_current_pos = _point_from_payload(nat_entry.get("current_pos"))
    nat_owned = bool(nat_entry.get("owned", False) or nat_entry.get("townhall_tag"))
    nat_flying = bool(nat_entry.get("is_flying", False))
    nat_taken = bool(nat_state in {"ESTABLISHED", "LANDED_UNSAFE", "SECURING"} and not nat_flying)
    nat_offsite = bool(nat_owned and not nat_taken)

    enemy_build_snapshot = awareness.mem.get(K("enemy", "build", "snapshot"), now=now, default={}) or {}
    if not isinstance(enemy_build_snapshot, dict):
        enemy_build_snapshot = {}
    bases_now = int(getattr(attention.macro, "bases_total", 0) or 0)
    enemy_bases_visible = int(enemy_build_snapshot.get("bases_visible", 0) or 0)
    enemy_base_gap = max(0, int(enemy_bases_visible) - int(bases_now))
    enemy_macro_lead_visible = bool(
        int(bases_now) < 2
        and int(enemy_bases_visible) >= int(cfg.enemy_macro_lead_visible_bases)
        and int(enemy_base_gap) >= int(cfg.enemy_macro_lead_gap)
    )
    delayed_natural_alarm = bool(
        awareness.mem.get(K("strategy", "parity", "alarms", "delayed_natural"), now=now, default=False)
    )

    rush_state = str(awareness.mem.get(K("enemy", "rush", "state"), now=now, default="NONE") or "NONE").upper()
    rush_active = rush_state in {"SUSPECTED", "CONFIRMED", "HOLDING"}
    last_confirmed_t = float(
        awareness.mem.get(K("enemy", "rush", "last_confirmed_t"), now=now, default=0.0) or 0.0
    )
    ended_t = float(awareness.mem.get(K("enemy", "rush", "ended_t"), now=now, default=0.0) or 0.0)
    last_pressure_t = float(
        awareness.mem.get(K("enemy", "rush", "last_seen_pressure_t"), now=now, default=0.0) or 0.0
    )
    rush_recent_mark = max(float(last_confirmed_t), float(ended_t))
    rush_recent = bool(rush_active or (rush_recent_mark > 0.0 and (float(now) - float(rush_recent_mark)) <= float(cfg.recent_window_s)))
    had_rush_context = bool(max(float(last_confirmed_t), float(ended_t), float(last_pressure_t)) > 0.0)
    clear_for = max(0.0, float(now) - float(last_pressure_t)) if float(last_pressure_t) > 0.0 else 9999.0

    own_total_power = _own_total_power(bot)
    own_main_power = _own_power_near(bot, center=bot.start_location, radius=18.0)
    own_nat_power = _own_power_near(bot, center=nat, radius=16.0)
    own_nat_bunker_count = 0
    own_nat_tank_count = 0
    for struct in list(getattr(bot, "structures", []) or []):
        try:
            if float(struct.distance_to(nat)) > 12.0:
                continue
            if struct.type_id == U.BUNKER and bool(getattr(struct, "is_ready", False)):
                own_nat_bunker_count += 1
        except Exception:
            continue
    for unit in list(getattr(bot, "units", []) or []):
        try:
            if float(unit.distance_to(nat)) > 13.0:
                continue
            if unit.type_id in {U.SIEGETANK, U.SIEGETANKSIEGED} and bool(getattr(unit, "is_ready", False)):
                own_nat_tank_count += 1
        except Exception:
            continue
    fortified = bool(own_nat_bunker_count > 0 or own_nat_tank_count > 0 or own_nat_power >= 7.5)
    no_direct_pressure = bool(not main_threat_active and len(main_enemy) <= 0 and int(attention.combat.primary_enemy_count) <= 0)

    nat_scan_t = float(awareness.mem.get(K("intel", "scan", "by_label", "enemy_natural", "last_t"), now=now, default=0.0) or 0.0)
    main_scan_t = float(awareness.mem.get(K("intel", "scan", "by_label", "enemy_main", "last_t"), now=now, default=0.0) or 0.0)
    nat_scan_stale = (float(now) - float(nat_scan_t)) >= float(cfg.rush_refresh_scan_stale_s)
    main_scan_stale = (float(now) - float(main_scan_t)) >= float(cfg.rush_refresh_scan_stale_s)
    natural_seen_on_ground = bool(getattr(attention.enemy_build, "enemy_natural_on_ground", False))
    enemy_presence_snapshot = awareness.mem.get(
        K("enemy", "army", "positions", "snapshot"),
        now=now,
        default={},
    ) or {}
    if not isinstance(enemy_presence_snapshot, dict):
        enemy_presence_snapshot = {}
    presence_nat_side_power = float(enemy_presence_snapshot.get("nat_side_power", 0.0) or 0.0)
    presence_state = str(enemy_presence_snapshot.get("state", "UNKNOWN") or "UNKNOWN")
    presence_primary = enemy_presence_snapshot.get("primary")
    presence_primary_side = ""
    if isinstance(presence_primary, dict):
        presence_primary_side = str(presence_primary.get("side", "") or "")

    refresh_label = ""
    refresh_target = None
    if (rush_recent or enemy_macro_lead_visible) and no_direct_pressure:
        if not natural_seen_on_ground and nat_scan_stale:
            refresh_label = "enemy_natural"
            refresh_target = getattr(attention.enemy_build, "enemy_natural_pos", None)
        elif main_scan_stale:
            refresh_label = "enemy_main"
            refresh_target = getattr(attention.enemy_build, "enemy_main_pos", None)
        elif nat_scan_stale:
            refresh_label = "enemy_natural"
            refresh_target = getattr(attention.enemy_build, "enemy_natural_pos", None)

    if refresh_target is None:
        refresh_label = ""

    # ---------------------------------------------------------------------------
    # should_secure e safe_to_land como FATOS observáveis simplificados.
    # A DECISÃO de postura (quando/como agir) está em army_posture_intel.
    # Esses valores são mantidos apenas para compatibilidade com consumidores
    # legados (LandBaseTask, DefenseBunkerTask, etc.).
    # ---------------------------------------------------------------------------

    # should_secure: fato — "há contexto para guarnecer a nat"
    # Critério simples: rush/pressão/alarme + nat não tomada + army mínima + nat não muito quente
    should_secure = bool(
        not nat_taken
        and bool(rush_recent or had_rush_context or enemy_macro_lead_visible or delayed_natural_alarm or nat_offsite)
        and bool(no_direct_pressure)
        and float(own_total_power) >= max(3.5, float(cfg.min_total_power) - 2.5)
        and float(nat_enemy_power) <= float(cfg.max_enemy_power_at_nat)
        and float(presence_nat_side_power) <= float(cfg.max_enemy_presence_power_at_nat)
    )

    # safe_to_land: fato — "o terreno permite pousar agora"
    nat_clear_to_land = bool(
        float(nat_enemy_power) <= min(float(cfg.max_enemy_power_at_nat), 1.1)
        and float(presence_nat_side_power) <= min(float(cfg.max_enemy_presence_power_at_nat), 1.0)
        and bool(no_direct_pressure)
    )
    local_nat_cover_ready = bool(
        own_nat_bunker_count > 0
        or own_nat_tank_count > 0
        or float(own_nat_power) >= float(cfg.natural_release_min_own_nat_power)
    )
    natural_release_window = bool(
        bool(no_direct_pressure)
        and float(clear_for) >= float(cfg.natural_release_clear_s)
        and bool(local_nat_cover_ready)
    )
    if bool(nat_taken):
        safe_to_land = True
    elif bool(nat_offsite) and bool(nat_clear_to_land):
        # Offsite com terreno limpo pode pousar
        safe_to_land = bool(fortified or bool(nat_clear_to_land))
    elif bool(should_secure) and bool(nat_clear_to_land):
        safe_to_land = bool(fortified or natural_release_window)
    else:
        safe_to_land = False

    defer_reason = "posture_delegated_to_army_posture_intel"
    if nat_offsite and not bool(safe_to_land) and bool(should_secure):
        defer_reason = "secure_before_landing"

    payload = {
        "updated_at": float(now),
        "rush_state": str(rush_state),
        "rush_active": bool(rush_active),
        "rush_recent": bool(rush_recent),
        "had_rush_context": bool(had_rush_context),
        "clear_for": float(clear_for),
        "nat_taken": bool(nat_taken),
        "nat_owned": bool(nat_owned),
        "nat_offsite": bool(nat_offsite),
        "nat_state": str(nat_state),
        "nat_current_pos": _point_payload(nat_current_pos),
        "bases_now": int(bases_now),
        "enemy_bases_visible": int(enemy_bases_visible),
        "enemy_base_gap": int(enemy_base_gap),
        "enemy_macro_lead_visible": bool(enemy_macro_lead_visible),
        "delayed_natural_alarm": bool(delayed_natural_alarm),
        "main_under_pressure": bool(not no_direct_pressure),
        "main_threat_active": bool(main_threat_active),
        "enemy_main_count": int(len(main_enemy)),
        "enemy_nat_count": int(len(nat_enemy)),
        "enemy_nat_power": float(round(float(nat_enemy_power), 3)),
        "enemy_presence_state": str(presence_state),
        "enemy_presence_nat_side_power": float(round(float(presence_nat_side_power), 3)),
        "enemy_presence_primary_side": str(presence_primary_side),
        "own_total_power": float(round(float(own_total_power), 3)),
        "own_main_power": float(round(float(own_main_power), 3)),
        "own_nat_power": float(round(float(own_nat_power), 3)),
        "fortified": bool(fortified),
        "local_nat_cover_ready": bool(local_nat_cover_ready),
        "natural_release_window": bool(natural_release_window),
        "own_nat_bunker_count": int(own_nat_bunker_count),
        "own_nat_tank_count": int(own_nat_tank_count),
        "should_secure": bool(should_secure),
        "safe_to_land": bool(safe_to_land),
        "defer_reason": str(defer_reason),
        "need_refresh": bool(refresh_label),
        "refresh_label": str(refresh_label),
        "refresh_target": _point_payload(refresh_target),
        "refresh_min_interval_s": float(cfg.rush_refresh_min_interval_s),
        "target": _point_payload(nat),
        "staging": _point_payload(staging_pos),
        "hold": _point_payload(hold_pos),
        "main_ramp_bunker_pos": _point_payload(_main_ramp_bunker_target(bot)),
    }
    awareness.mem.set(
        K("intel", "map_control", "our_nat", "snapshot"),
        value=dict(payload),
        now=now,
        ttl=float(cfg.ttl_s),
    )
    awareness.mem.set(
        K("intel", "map_control", "our_nat", "should_secure"),
        value=bool(should_secure),
        now=now,
        ttl=float(cfg.ttl_s),
    )
    awareness.mem.set(
        K("intel", "map_control", "our_nat", "safe_to_land"),
        value=bool(safe_to_land),
        now=now,
        ttl=float(cfg.ttl_s),
    )
    if nat_entry:
        nat_entry["safe_to_land"] = bool(safe_to_land)
        nat_entry["safe_to_mine"] = bool(nat_taken or safe_to_land)
        nat_entry["fortified"] = bool(fortified)
        nat_entry["updated_at"] = float(now)
        our_bases["NATURAL"] = dict(nat_entry)
        awareness.mem.set(K("intel", "our_bases", "registry"), value=dict(our_bases), now=now, ttl=8.0)
    awareness.mem.set(
        K("intel", "map_control", "our_nat", "last_update_t"),
        value=float(now),
        now=now,
        ttl=None,
    )
    awareness.mem.set(
        K("intel", "placements", "main_ramp_bunker"),
        value=_point_payload(_main_ramp_bunker_target(bot)),
        now=now,
        ttl=float(cfg.ttl_s),
    )
